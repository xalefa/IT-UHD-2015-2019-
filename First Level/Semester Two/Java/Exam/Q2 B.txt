
package midtermexam;

import java.util.Arrays;
import java.util.Scanner;

/**
 *
 * @author Govar
 * Template  B:
 * Q2/
 * write a program using switch case for english vowels, if the letter is the 
 * vowel print "vowel letter" else print "not vowel".
 */
public class MidTermExam {

 
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);

        //Answer :
        System.out.println("please enter a letter");
        char letter=input.next().charAt(0);
        switch (letter) {
            case 'a':
            case 'e':
            case 'i':
            case 'o':
            case 'u':System.out.println("vowel letter");
                break;
            default:
                System.out.println("not vowel");
        }
    }
}
